Dependency Injection

Decorators 

import {inject} from 'aurelia-framework';
import {HttpClient} from 'aurelia-http-client';

@inject(HttpClient)
export class App {
    constructor(http) {
        this.http = http;
    }
}
Static member 

import {HttpClient} from 'aurelia-http-client';

export class App {
    static inject = [HttpClient];

    constructor(http) {
        this.http = http;
    }
}
