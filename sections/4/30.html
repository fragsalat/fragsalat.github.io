Routing

Configuration 

export class App {
    configureRouter(config, router) {
        config.title = 'Aurelia';
        config.map([{
            route: ['', 'welcome'],
            name: 'welcome',
            moduleId: 'welcome',
            nav: true,
            title: 'Welcome'
        }]);

        this.router = router;
    }
}

import {Router} from 'aurelia-framework';

export class Users {
	static inject = [Router];
	constructor(router) {
		this.router = router;
	}
	onClick(id) {
		this.router.navigateToRoute('welcome', id);
	}
}
